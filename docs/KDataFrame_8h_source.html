43KVector StringTo(const std::string &inp_val)
44{
45 // 正規表現
46 std::regex IntPattern("^-?\\d+$"); // 整数
47 std::regex FloatPattern("^-?\\d*\\.\\d+$"); // 浮動小数点
48 std::regex ExpPattern("^-?\\d*\\.?\\d+[eE][-+]?\\d+$"); // 指数表記
49
50 if (std::regex_match(inp_val, IntPattern))
51 {
52 return std::stoi(inp_val); // 整数に変換
53 }
54 else if (std::regex_match(inp_val, FloatPattern))
55 {
56 return std::stod(inp_val); // 浮動小数点数に変換
57 }
58 else if (std::regex_match(inp_val, ExpPattern))
59 {
60 return std::stod(inp_val); // 指数表記を浮動小数点数に変換
61 }
62 else return inp_val; // どれにも該当しない場合は文字列として返す
63}
